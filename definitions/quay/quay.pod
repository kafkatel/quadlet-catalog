[Unit]
Description=Pod for the Quay Container Registry Stack
After=network-online.target quay-network-network.service
Wants=network-online.target
Requires=quay-network-network.service

[Pod]
Network=quay-network.network

# Quay HTTP and HTTPS ports.
# Change to 80:8080 and 443:8443 for standard ports (requires root).
# For reverse proxy deployments, add the proxy to the quay-backend network
# and remove these PublishPort lines so Quay is only reachable via the proxy.
PublishPort=8080:8080
PublishPort=8443:8443

[Install]
WantedBy=default.target
